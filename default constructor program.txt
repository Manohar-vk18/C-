#include <iostream>
using namespace std;

class teacher
{

    string name;
    string dept;
    int exp;
    double salary;

   //constructor;
   teacher()
   {
       name="Ohm";
       dept="CS";
       exp=5;

        cout<<"Teacher salary = "<<salary<<endl;
    }
};
 int main()
 {
     teacher t1;
     t1.displayinfo();
     
     
     return 0;
 }
